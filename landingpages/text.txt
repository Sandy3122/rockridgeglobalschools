      });
    }, observerOptions);

      observer.observe(section);

      const heroImage = document.querySelector('.hero-main-image');
      const nav = document.querySelector('.topnav');
      
      if (heroImage && scrolled < 500) {
        heroImage.style.transform = `translateY(${scrolled * 0.2}px)`;
      }
      
      if (nav) {
        if (scrolled > 50) {
